
create table Account (
ID int,
UUID char (30),
Locale char (30),
LastLogin datetime,
TimeZone char (30),
isAdmin int,
Primary key (ID)
);

create table Map (
ID int,
MapOriginal char (100),
MapGrid char (100),
Owner int not null references Account(ID),
Primary key (ID)
);

create table Game (
ID int,
DM int not null references Account (ID),
Map int not null references Map (ID),
Notes varchar(2000),
Password char (30),
ChatHistory char (100),
Primary key (ID)
);
